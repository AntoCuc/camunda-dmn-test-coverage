26      @Rule

29      @Test
30      public void testBronze() {
31          DmnEngine dmnEngine = dmnEngineRule.getDmnEngine();
32          InputStream inputStream = SampleUsageTest.class.getResourceAsStream("Example.dmn");
33  
34          VariableMap variables = Variables.createVariables();
35          variables.put("status", "bronze");
36          variables.put("sum", 0);
37  
38          DmnDecision decision = dmnEngine.parseDecision("orderDecision", inputStream);
39          DmnDecisionResult result = dmnEngine.evaluateDecision(decision, variables);
40  
41          assertEquals(1, result.getResultList().size());
42          assertEquals("notok", result.getSingleResult().get("result"));
43          assertEquals("work on your status first, as bronze you're not going to get anything", result.getSingleResult().get("reason"));
44  
45          assertEquals(0.25, dmnEngineRule.getCoverage(), 0.0);
46          assertEquals(singletonList("rule1"), dmnEngineRule.getMatchedRules());
47      }
